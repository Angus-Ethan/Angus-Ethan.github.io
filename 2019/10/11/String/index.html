String

C++ 标准库提供了 string 类类型，使用时需要#include将string类型包含到程序中

string类中的四个缺省方法

构造函数

MyString(char *data = nullptr)
{
    if(data == nullptr)
    {
        m_data = new char[1];
        assert(m_data != nullptr);
        *m_data = '\0';
    }
    else
    {
        m_data = new char[strlen(data) + 1];
        assert(m_data != nullptr);
        strcpy(m_data, data);
    }
}

拷贝构造函数

MyString(const MyString &other)
{
    m_data = new char[strlen(other.m_data) + 1];
    assert(m_data != nullptr);
    strcpy(m_data, other.m_data);
}

析构函数

~MyString()
{
    if(m_data != nullptr)
    {
        delete[] m_data;
        m_data = nullptr;
    }
}

赋值运算符

const MyString& operator=(const MyString &other)
{
    if(this == &other){
        return *this;
    }
    delete[] m_data;
    m_data = new char[strlen(other.m_data) + 1];
    assert(m_data != nullptr);
    strcpy(m_data, other.m_data);
    return *this;
}

string类的常用方法

复制函数

char* MyStrCpy(char *pDest, const char *pSrc)
{
    if(pDest == nullptr || pSrc == nullptr)
    {
        return nullptr;
    }
    if(pDest == pSrc)
    {
        return pDest;
    }
    char *pIter = pDest;
    while((*pIter++ = *pSrc++) != '\0');
    return pDest;
}

字符串拼接

char* MyStrCat(char *pDest, const char *pSrc)
{
    if(pDest == nullptr || pSrc == nullptr)
    {
        return nullptr;
    }
    char *pIter = pDest + strlen(pDest);
    while((*pIter++ = *pSrc++) != '\0');
    return pDest;
}
